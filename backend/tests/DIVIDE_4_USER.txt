DivideEnCuatro(n)
begin
    If (n > 1) then
    begin
        cuarto <- n / 4;
        
        // 4 llamadas recursivas
        CALL DivideEnCuatro(cuarto);
        CALL DivideEnCuatro(cuarto);
        CALL DivideEnCuatro(cuarto);
        CALL DivideEnCuatro(cuarto);
        
        // Combinacion lineal
        i <- 0;
        while (i < n) do
        begin
            i <- i + 1;
        end;
    end;
end
